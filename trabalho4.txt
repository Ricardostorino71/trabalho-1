uma estrutura de repetição; 

# Loop while: imprime os números de 1 a 5
print("Loop usando while:")
num = 1
while num <= 5:
    print(num)
    num += 1

# Loop for: imprime os elementos de uma lista
print("\nLoop usando for:")
lista = ["a", "b", "c", "d", "e"]
for elemento in lista:
    print(elemento)
